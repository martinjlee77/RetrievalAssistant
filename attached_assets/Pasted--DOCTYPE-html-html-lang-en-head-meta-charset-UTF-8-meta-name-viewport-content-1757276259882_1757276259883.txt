    <script>
        // Check authentication and load dashboard
        window.addEventListener('load', function() {
            const token = localStorage.getItem('authToken');
            
            if (!token) {
                // Redirect to login if not authenticated
                window.location.href = 'login.html';
                return;
            }
            
            loadDashboard();
        });
        
        async function loadDashboard() {
            try {
                const token = localStorage.getItem('authToken');
                
                // Construct API URL for Replit environment
                const apiUrl = window.location.protocol + '//' + window.location.hostname + ':3001/api/user/profile';
                const response = await fetch(apiUrl, {
                    headers: {
                        'Authorization': `Bearer ${token}`
                    }
                });
                
                if (!response.ok) {
                    throw new Error('Failed to load profile');
                }
                
                const data = await response.json();
                populateDashboard(data);
                
            } catch (error) {
                console.error('Dashboard load error:', error);
                
                // If authentication fails, redirect to login
                if (error.message.includes('authentication') || error.message.includes('token')) {
                    localStorage.removeItem('authToken');
                    localStorage.removeItem('userData');
                    window.location.href = 'login.html?message=Session expired. Please sign in again.&type=error';
                } else {
                    alert('Failed to load dashboard. Please refresh the page or contact support.');
                }
            }
        }
        
        function populateDashboard(data) {
            const user = data.user;
            const analyses = data.recent_analyses || [];
            const transactions = data.transactions || [];
            
            // Update user information
            document.getElementById('userName').textContent = user.first_name;
            document.getElementById('memberSince').textContent = `Member since ${new Date(user.member_since).toLocaleDateString()}`;
            document.getElementById('userEmail').textContent = user.email;
            document.getElementById('userCompany').textContent = user.company_name;
            document.getElementById('userTitle').textContent = user.job_title;
            
            // Update credits and usage
            document.getElementById('freeAnalysesCount').textContent = user.free_analyses_remaining;
            document.getElementById('creditsBalance').textContent = user.credits_balance;
            document.getElementById('totalAnalyses').textContent = analyses.length;
            
            // Calculate total spent
            const totalSpent = transactions
                .filter(t => t.reason === 'analysis_charge')
                .reduce((sum, t) => sum + Math.abs(t.amount), 0);
            document.getElementById('totalSpent').textContent = `$${totalSpent}`;
            
            // Hide free analyses section if no free analyses remaining
            if (user.free_analyses_remaining === 0) {
                document.getElementById('freeAnalysesSection').style.display = 'none';
            }
            
            // Populate recent analyses
            const analysesContainer = document.getElementById('recentAnalyses');
            
            if (analyses.length === 0) {
                analysesContainer.innerHTML = `
                    <div class="empty-state">
                        <div class="icon">ðŸ“„</div>
                        <p>No analyses yet. Ready to get started?</p>
                        <a href="https://a45dfa8e-cff4-4d5e-842f-dc8d14b3b2d2-00-3khkzanf4tnm3.picard.replit.dev/" class="btn btn-primary" target="_blank">Start Your First Analysis</a>
                    </div>
                `;
            } else {
                analysesContainer.innerHTML = analyses.map(analysis => `
                    <div class="analysis-item">
                        <div class="analysis-info">
                            <h4>${analysis.asc_standard} Analysis</h4>
                            <p>${new Date(analysis.completed_at).toLocaleDateString()}</p>
                        </div>
                        <div class="analysis-cost">
                            ${analysis.cost === 0 ? 'FREE' : `$${analysis.cost}`}
                        </div>
                    </div>
                `).join('');
            }
            
            // Show main content and hide loading
            document.getElementById('loadingState').style.display = 'none';
            document.getElementById('mainContent').style.display = 'block';
        }
        
        function logout() {
            localStorage.removeItem('authToken');
            localStorage.removeItem('userData');
            window.location.href = 'login.html?message=You have been signed out successfully.&type=success';
        }
        
        function contactSupport() {
            alert('To add credits to your account, please contact our support team at support@veritaslogic.ai or use the contact form. We\'ll process your request within 24 hours.');
        }

        document.querySelector('.mobile-menu-btn').addEventListener('click', function() {

        });
